### 피보나치
```c
//fivo(0) = 0
//fivo(1) = 1
//fivo(2) = fivo(0) + fivo(1)
//fivo(n) = fivo(n-2) + fivo(n-1)

//피보나치 x번째 값을 리턴 (0부터 시작) 
int fivo(int x){
	if(x == 0)
		return 0;
	else if(x == 1)
		return 1;
	else
		return fivo(x-2) + fivo(x-1);
	
}

void main(){

	int i; //for문을 위해서 i지정
	
	//피보나치 10개의 값 출력 
	for(i=0; i<10; i++){
		printf("%3d", fivo(i));	
	}
}
```
